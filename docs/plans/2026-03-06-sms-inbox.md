# SMS Business Inbox Implementation Plan

> **For Claude:** REQUIRED SUB-SKILL: Use superpowers:executing-plans to implement this plan task-by-task.

**Goal:** Build a threaded SMS inbox in the existing admin dashboard so Gregg can receive customer texts and reply from his phone, with every reply correctly routed back to the right customer through the business number.

**Architecture:** Twilio stays as pure transport. A new `smsThreads` Firestore collection groups messages by customer phone. The inbound webhook creates/updates threads and sends Gregg a notification SMS with a direct link to the thread. Gregg replies via a new admin UI page; the backend loads the thread, gets the customer phone, and sends via Twilio. No Twilio Conversations needed.

**Tech Stack:** Next.js App Router, Firebase Admin SDK (Firestore), Twilio SDK (existing `sendSms`), Vitest (tests), existing `requireAdmin` auth pattern.

// src/lib/services/sms-thread-service.ts
import { getAdminDb } from '@/lib/utils/firebase-admin';
import { FieldValue } from 'firebase-admin/firestore';

const THREADS_COLLECTION = 'smsThreads';
const MESSAGES_COLLECTION = 'smsMessages';

export interface SmsThread {
  id: string;
  customerPhone: string;
  customerName?: string | null;
  status: 'open' | 'closed';
  lastMessageAt: FirebaseFirestore.Timestamp | null;
  lastMessagePreview: string;
  unreadCount: number;
  lastInboundAt?: FirebaseFirestore.Timestamp | null;
  lastOutboundAt?: FirebaseFirestore.Timestamp | null;
  createdAt: FirebaseFirestore.Timestamp | null;
  updatedAt: FirebaseFirestore.Timestamp | null;
}

export interface SmsThreadMessage {
  id: string;
  threadId: string;
  direction: 'inbound' | 'outbound';
  senderType: 'customer' | 'admin' | 'system';
  from: string;
  to: string;
  body: string;
  createdAt: string | null;
  twilioMessageSid?: string | null;
}

function toIso(ts: FirebaseFirestore.Timestamp | null | undefined): string | null {
  return ts?.toDate?.()?.toISOString?.() ?? null;
}

/**
 * Find an open thread for this customer phone, or create one.
 * Returns { threadId, created }.
 */
export async function findOrCreateThread(
  customerPhone: string
): Promise<{ threadId: string; created: boolean }> {
  const db = getAdminDb();
  const snapshot = await db
    .collection(THREADS_COLLECTION)
    .where('customerPhone', '==', customerPhone)
    .where('status', '==', 'open')
    .limit(1)
    .get();

  if (!snapshot.empty) {
    return { threadId: snapshot.docs[0].id, created: false };
  }

  const ref = await db.collection(THREADS_COLLECTION).add({
    customerPhone,
    customerName: null,
    status: 'open',
    lastMessageAt: FieldValue.serverTimestamp(),
    lastMessagePreview: '',
    unreadCount: 0,
    lastInboundAt: null,
    lastOutboundAt: null,
    createdAt: FieldValue.serverTimestamp(),
    updatedAt: FieldValue.serverTimestamp(),
  });

  return { threadId: ref.id, created: true };
}

/**
 * Update thread metadata when a customer message arrives.
 */
export async function updateThreadOnInbound(
  threadId: string,
  messageBody: string
): Promise<void> {
  const db = getAdminDb();
  const preview = messageBody.slice(0, 80);
  await db.collection(THREADS_COLLECTION).doc(threadId).update({
    lastMessageAt: FieldValue.serverTimestamp(),
    lastMessagePreview: preview,
    lastInboundAt: FieldValue.serverTimestamp(),
    unreadCount: FieldValue.increment(1),
    updatedAt: FieldValue.serverTimestamp(),
  });
}

/**
 * Update thread metadata when Gregg sends a reply.
 */
export async function updateThreadOnOutbound(
  threadId: string,
  messageBody: string
): Promise<void> {
  const db = getAdminDb();
  await db.collection(THREADS_COLLECTION).doc(threadId).update({
    lastMessageAt: FieldValue.serverTimestamp(),
    lastMessagePreview: `You: ${messageBody.slice(0, 74)}`,
    lastOutboundAt: FieldValue.serverTimestamp(),
    updatedAt: FieldValue.serverTimestamp(),
  });
}

/**
 * Reset unread count when Gregg opens the thread.
 */
export async function markThreadRead(threadId: string): Promise<void> {
  const db = getAdminDb();
  await db.collection(THREADS_COLLECTION).doc(threadId).update({
    unreadCount: 0,
    updatedAt: FieldValue.serverTimestamp(),
  });
}

/**
 * Get all open threads, sorted by most recent message.
 */
export async function getThreads(): Promise<SmsThread[]> {
  const db = getAdminDb();
  const snapshot = await db
    .collection(THREADS_COLLECTION)
    .orderBy('lastMessageAt', 'desc')
    .limit(100)
    .get();

  return snapshot.docs.map(doc => {
    const d = doc.data();
    return {
      id: doc.id,
      customerPhone: d.customerPhone,
      customerName: d.customerName ?? null,
      status: d.status,
      lastMessageAt: d.lastMessageAt,
      lastMessagePreview: d.lastMessagePreview ?? '',
      unreadCount: d.unreadCount ?? 0,
      lastInboundAt: d.lastInboundAt ?? null,
      lastOutboundAt: d.lastOutboundAt ?? null,
      createdAt: d.createdAt ?? null,
      updatedAt: d.updatedAt ?? null,
    };
  });
}

/**
 * Get a single thread by ID.
 */
export async function getThread(threadId: string): Promise<SmsThread | null> {
  const db = getAdminDb();
  const doc = await db.collection(THREADS_COLLECTION).doc(threadId).get();
  if (!doc.exists) return null;
  const d = doc.data()!;
  return {
    id: doc.id,
    customerPhone: d.customerPhone,
    customerName: d.customerName ?? null,
    status: d.status,
    lastMessageAt: d.lastMessageAt,
    lastMessagePreview: d.lastMessagePreview ?? '',
    unreadCount: d.unreadCount ?? 0,
    lastInboundAt: d.lastInboundAt ?? null,
    lastOutboundAt: d.lastOutboundAt ?? null,
    createdAt: d.createdAt ?? null,
    updatedAt: d.updatedAt ?? null,
  };
}

/**
 * Get messages for a thread, oldest-first for display.
 */
export async function getThreadMessages(threadId: string): Promise<SmsThreadMessage[]> {
  const db = getAdminDb();
  const snapshot = await db
    .collection(MESSAGES_COLLECTION)
    .where('threadId', '==', threadId)
    .orderBy('createdAt', 'asc')
    .limit(200)
    .get();

  return snapshot.docs.map(doc => {
    const d = doc.data();
    return {
      id: doc.id,
      threadId: d.threadId,
      direction: d.direction,
      senderType: d.senderType ?? (d.direction === 'inbound' ? 'customer' : 'admin'),
      from: d.from,
      to: d.to,
      body: d.body,
      createdAt: toIso(d.createdAt),
      twilioMessageSid: d.twilioMessageSid ?? null,
    };
  });
}

// src/app/api/twilio/incoming-sms/route.ts
import { NextRequest, NextResponse } from 'next/server';
import twilio from 'twilio';
import { sendSms } from '@/lib/services/twilio-service';
import { saveSmsMessage } from '@/lib/services/sms-message-service';
import { findOrCreateThread, updateThreadOnInbound } from '@/lib/services/sms-thread-service';

const authToken = process.env.TWILIO_AUTH_TOKEN;
const forwardToNumber = process.env.GREGG_SMS_FORWARD_NUMBER || process.env.ADMIN_FORWARD_SMS_TO;
const baseUrl = process.env.NEXT_PUBLIC_BASE_URL || 'https://fairfieldairportcar.com';

const TWIML_OK = '<?xml version="1.0" encoding="UTF-8"?><Response></Response>';

export async function POST(request: NextRequest) {
  const rawBody = await request.text();
  const signature = request.headers.get('x-twilio-signature') || '';
  const url = request.url;

  if (!authToken) {
    console.error('[Twilio webhook] TWILIO_AUTH_TOKEN not set');
    return new NextResponse('Server configuration error', { status: 500 });
  }

  const params: Record<string, string> = {};
  new URLSearchParams(rawBody).forEach((value, key) => { params[key] = value; });

  const isValid = twilio.validateRequest(authToken, signature, url, params);
  if (!isValid) {
    console.warn('[Twilio webhook] Invalid signature');
    return new NextResponse('Forbidden', { status: 403 });
  }

  const from = params.From || '';
  const to = params.To || '';
  const body = params.Body || '';
  const messageSid = params.MessageSid || '';

  // Find or create thread for this customer
  let threadId: string | null = null;
  try {
    const result = await findOrCreateThread(from);
    threadId = result.threadId;
    await updateThreadOnInbound(threadId, body);
  } catch (err) {
    console.error('[Twilio webhook] Thread routing failed:', err);
  }

  // Save message to Firestore
  await saveSmsMessage({
    from,
    to,
    body,
    direction: 'inbound',
    twilioMessageSid: messageSid,
    threadId: threadId ?? undefined,
    senderType: 'customer',
  });

  // Notify Gregg with deep link to the specific thread
  if (forwardToNumber && body.trim() && threadId) {
    try {
      const threadUrl = `${baseUrl}/admin/messages/${threadId}`;
      const preview = body.length > 60 ? body.slice(0, 60) + '…' : body;
      const notificationBody = `New message from ${from}: "${preview}"\nReply: ${threadUrl}`;
      await sendSms({ to: forwardToNumber, body: notificationBody });
    } catch (err) {
      // Never fail the webhook because notification failed
      console.error('[Twilio webhook] Failed to notify Gregg:', err);
    }
  }

  return new NextResponse(TWIML_OK, {
    status: 200,
    headers: { 'Content-Type': 'text/xml' },
  });
}

// src/app/api/admin/messages/send/route.ts
import { NextResponse } from 'next/server';
import { z } from 'zod';
import { requireAdmin } from '@/lib/utils/auth-server';
import { getThread, updateThreadOnOutbound } from '@/lib/services/sms-thread-service';
import { sendSms } from '@/lib/services/twilio-service';
import { saveSmsMessage } from '@/lib/services/sms-message-service';

const schema = z.object({
  threadId: z.string().min(1),
  body: z.string().min(1),
});

const businessNumber = process.env.TWILIO_PHONE_NUMBER || '';

export async function POST(request: Request) {
  const authResult = await requireAdmin(request);
  if (!authResult.ok) return authResult.response;

  const raw = await request.json().catch(() => ({}));
  const parsed = schema.safeParse(raw);
  if (!parsed.success) {
    return NextResponse.json({ error: 'threadId and body are required' }, { status: 400 });
  }

  const { threadId, body } = parsed.data;
  const trimmedBody = body.trim();
  if (!trimmedBody) {
    return NextResponse.json({ error: 'Message body cannot be empty' }, { status: 400 });
  }

  const thread = await getThread(threadId);
  if (!thread) {
    return NextResponse.json({ error: 'Thread not found' }, { status: 404 });
  }

  // Send via Twilio from business number to customer
  const message = await sendSms({ to: thread.customerPhone, body: trimmedBody });

  // Save outbound to Firestore
  await saveSmsMessage({
    from: businessNumber,
    to: thread.customerPhone,
    body: trimmedBody,
    direction: 'outbound',
    senderType: 'admin',
    threadId,
    twilioMessageSid: (message as any)?.sid ?? undefined,
  });

  // Update thread metadata
  await updateThreadOnOutbound(threadId, trimmedBody);

  return NextResponse.json({ success: true });
}

function formatRelativeTime(iso: string | null): string {
  if (!iso) return '';
  const date = new Date(iso);
  const now = new Date();
  const diffMs = now.getTime() - date.getTime();
  const diffMins = Math.floor(diffMs / 60000);
  if (diffMins < 1) return 'just now';
  if (diffMins < 60) return `${diffMins}m ago`;
  const diffHours = Math.floor(diffMins / 60);
  if (diffHours < 24) return `${diffHours}h ago`;
  return date.toLocaleDateString('en-US', { month: 'short', day: 'numeric' });
}

  "I'm on my way",
  "Please confirm your pickup address",

function formatTime(iso: string | null): string {
  if (!iso) return '';
  return new Date(iso).toLocaleTimeString('en-US', { hour: 'numeric', minute: '2-digit' });
}

export default function ThreadDetailPage() {
  const { threadId } = useParams<{ threadId: string }>();
  const router = useRouter();
  const [thread, setThread] = useState<Thread | null>(null);
  const [messages, setMessages] = useState<Message[]>([]);
  const [loading, setLoading] = useState(true);
  const [error, setError] = useState<string | null>(null);
  const [replyBody, setReplyBody] = useState('');
  const [sending, setSending] = useState(false);
  const [sendError, setSendError] = useState<string | null>(null);
  const bottomRef = useRef<HTMLDivElement>(null);

  const loadThread = useCallback(() => {
    authFetch(`/api/admin/messages/threads/${threadId}`)
      .then(res => { if (!res.ok) throw new Error('Failed to load thread'); return res.json(); })
      .then(data => {
        setThread(data.thread);
        setMessages(data.messages ?? []);
        // Mark as read
        authFetch(`/api/admin/messages/threads/${threadId}/read`, { method: 'POST' }).catch(() => {});

      setSending(false);

            <div style={{ fontSize: '13px', color: '#6b7280' }}>{thread.customerPhone}</div>
